Variable

			Var Name : 
						fullName  -> Camel Case   :  We Used mostly this
						full_name -> Snake Case
						full-name -> kaba Case     : Not allow
						FullName  -> Pascal Case
			
			Var :  Var can be re-declared & up dated. A global scope var. 
					var name = 'demo';  .
					var name = "one";   .

			let : Var cannot be re-declared but can be updated. A block{fun} scope var.
					let age = 20;
					let age = 22;  xxxx
					age = 22;      .
					age = 50;      .

			const : var cannot be re-declared or updated. A block {fun} scope var.
					const no = 20; 
					const no = 25;  xxxx
					no = 23;        xxxx
